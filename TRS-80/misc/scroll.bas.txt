100 CLS:CLEAR 300
110 INPUT"MSG";M$:INPUT"WIDTH OF WINDOW";W
120 CLS:M$=M$+STRING$(W," ")
130 S=1:FOR T=1 TO LEN(M$)
140 FOR G=1 TO 50:NEXT
150 IF T<LEN(M$) THEN L=T :ELSE L=LEN(M$)
160 IF L>W THEN S=L-W+1:L=W
170 S$=MID$(M$,S,L)
180 PRINT @296-L,S$;:NEXT
190 GOTO 130
